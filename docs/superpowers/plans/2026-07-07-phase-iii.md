# 三期实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 sdd-harness 从插件型 SDD Harness 升级为 CLI-first、Agent-agnostic、codebase-memory-powered 的通用 SDD Agent Harness。

**Architecture:** 新增 CLI 作为唯一确定性入口，Core 暴露统一 SddCore.execute() 契约，agent-protocol 定义 Agent 交互协议，codebase-memory 内置托管 MCP，各 Agent Adapter 只做命令翻译。分 6 个阶段按 P0→P5 推进。

**Tech Stack:** TypeScript ESM, Node.js >= 22, Vitest, Zod, YAML, npm workspaces, npx (codebase-memory-mcp 托管)

## Global Constraints

- Node.js >= 22，所有包 `engines.node` 设为 `">=22"`
- 所有包版本统一 `0.1.0`
- TypeScript ESM，2 空格缩进
- 中文注释和文档，英文标识符/错误码/命令字面量
- TDD：先写测试，确认失败，再实现，确认通过，提交
- 不发布 npm，通过 `scripts/install.sh` / `scripts/install.ps1` 安装
- 禁止 shell=true、bash -c、eval，命令使用结构化形式 `{command, args}`
- 退出码必须等于 `CommandResult.exitCode`
- codebase-memory-mcp 不可用时禁止静默降级

```typescript
#!/usr/bin/env node
// packages/cli/src/cli.ts — sdd / sdd-harness CLI 入口
import { parseArgs } from "node:util";
import { ExitCode } from "./exit-codes.js";

const PKG_VERSION = "0.1.0";
const BIN_NAME = process.env.SDD_BIN_NAME ?? "sdd";

const HELP_TEXT = `${BIN_NAME} — SDD Agent Harness CLI

用法: ${BIN_NAME} <command> [options]

命令:
  init              初始化 .sdd/
  status            显示当前 SDD 状态
  new <需求>        创建新变更
  design            生成设计制品
  plan              生成实施计划
  build             构建 (build next / build complete)
  verify            验证
  review            审查
  archive           归档
  auto <需求>       自动推进完整流程

通用参数:
  --json            JSON 输出
  --cwd <path>      项目根目录 (默认当前目录)
  --change <id>     指定变更 ID
  --timeout <s>     超时秒数
  --non-interactive 禁止交互
  --force           强制
  --verbose         详细输出
  --help            帮助
  --version         版本
`;

const COMMANDS = [
  "init", "status", "new", "design", "plan",
  "build", "verify", "review", "archive", "auto",
];

function main(): void {
  const args = process.argv.slice(2);

  if (args.length === 0 || args.includes("--help") || args.includes("-h")) {
    console.log(HELP_TEXT);
    process.exit(ExitCode.SUCCESS);
  }

  if (args.includes("--version") || args.includes("-v")) {
    console.log(`${BIN_NAME} v${PKG_VERSION}`);
    process.exit(ExitCode.SUCCESS);
  }

  const command = args[0];
  if (!COMMANDS.includes(command)) {
    console.error(`未知命令: ${command}`);
    console.error(`可用命令: ${COMMANDS.join(", ")}`);
    process.exit(ExitCode.INVALID_ARGS);
  }

  // 后续阶段实现实际命令分发
  console.log(`Command: ${command} (not yet implemented)`);
  process.exit(ExitCode.SUCCESS);
}

main();
```

```typescript
// packages/cli/src/json-output.ts
import type { CommandResult } from "@sdd-harness/core";

/** 以 --json 格式输出 CommandResult 到 stdout */
export function outputJson(result: CommandResult): void {
  const json = {
    ok: result.ok,
    state: result.state,
    exitCode: result.exitCode,
    changeId: result.changeId,
    next: result.next,
    warnings: result.warnings,
    actionRequired: result.actionRequired,
    error: result.error,
  };
  console.log(JSON.stringify(json, null, 2));
}

/** 以人类可读格式输出 */
export function outputText(result: CommandResult): void {
  console.log(`State: ${result.state}`);
  if (result.changeId) console.log(`Change: ${result.changeId}`);
  if (result.next) console.log(`Next: ${result.next}`);
  if (result.warnings?.length) {
    for (const w of result.warnings) {
      console.log(`Warning [${w.code}]: ${w.message}`);
      if (w.next) console.log(`  → ${w.next}`);
    }
  }
  if (result.error) {
    console.error(`Error [${result.error.code}]: ${result.error.message}`);
    if (result.error.next) console.error(`  → ${result.error.next}`);
  }
}
```

```typescript
// packages/cli/src/cli.ts — 在 main() 中添加命令分发
import { Core } from "@sdd-harness/core";
import { ExitCode } from "./exit-codes.js";
import { outputJson, outputText } from "./json-output.js";
import { runInit } from "./commands/init.js";
import { runStatus } from "./commands/status.js";
import { runNew } from "./commands/new.js";
import { runDesign } from "./commands/design.js";
import { runPlan } from "./commands/plan.js";

async function main(): Promise<void> {
  const { values, positionals } = parseArgs({
    args: process.argv.slice(2),
    options: {
      json: { type: "boolean", default: false },
      cwd: { type: "string" },
      change: { type: "string" },
      timeout: { type: "string" },
      "non-interactive": { type: "boolean", default: false },
      force: { type: "boolean", default: false },
      verbose: { type: "boolean", default: false },
      help: { type: "boolean", default: false },
      version: { type: "boolean", default: false },
    },
    allowPositionals: true,
  });

  if (values.help) { console.log(HELP_TEXT); process.exit(ExitCode.SUCCESS); }
  if (values.version) { console.log(`${BIN_NAME} v${PKG_VERSION}`); process.exit(ExitCode.SUCCESS); }

  const [command] = positionals;
  if (!command || !COMMANDS.includes(command)) {
    console.error(`未知命令: ${command ?? "(空)"}`);
    console.error(`可用命令: ${COMMANDS.join(", ")}`);
    process.exit(ExitCode.INVALID_ARGS);
  }

  const core = new Core();
  const cwd = values.cwd ?? process.cwd();
  const json = values.json ?? false;

  let result: CommandResult;

  switch (command) {
    case "init":
      result = await runInit(core, cwd, {}, undefined);
      break;
    case "status":
      result = await runStatus(core, cwd, undefined);
      break;
    case "new": {
      const requirement = positionals.slice(1).join(" ") || "";
      result = await runNew(core, cwd, requirement, {}, undefined);
      break;
    }
    case "design":
      result = await runDesign(core, cwd, {}, undefined);
      break;
    case "plan":
      result = await runPlan(core, cwd, {}, undefined);
      break;
    default:
      console.log(`Command: ${command} (not yet implemented)`);
      process.exit(ExitCode.SUCCESS);
  }

  if (json) outputJson(result);
  else outputText(result);

  process.exit(result.exitCode);
}

main().catch((err) => {
  console.error("CLI 内部错误:", err.message);
  process.exit(ExitCode.GENERAL_ERROR);
});
```

```typescript
// packages/cli/src/commands/build.ts
import type { SddCore, CommandResult } from "@sdd-harness/core";
import fs from "node:fs/promises";

export async function runBuild(
  core: SddCore,
  cwd: string,
  subcommand: string | undefined,
  taskId: string | undefined,
  resultPath: string | undefined,
  args: Record<string, unknown>,
  signal?: AbortSignal,
): Promise<CommandResult> {
  if (subcommand === "next") {
    return core.execute({
      command: "build",
      cwd,
      args: { ...args, subcommand: "next" },
      signal,
    });
  }

  if (subcommand === "complete") {
    if (!taskId || !resultPath) {
      return {
        ok: false,
        state: "FAILED",
        exitCode: 2,
        error: {
          code: "E_INVALID_ARGS",
          message: "build complete 需要 --task 和 --result 参数",
        },
      };
    }
    const resultJson = JSON.parse(await fs.readFile(resultPath, "utf-8"));
    return core.execute({
      command: "build",
      cwd,
      args: { ...args, subcommand: "complete", taskId, result: resultJson },
      signal,
    });
  }

  // 无子命令：返回 build 状态
  return core.execute({ command: "build", cwd, args, signal });
}
```

```typescript
// packages/codebase-memory/src/lifecycle.ts
import { spawn, type ChildProcess } from "node:child_process";
import type { McpLifecycleResult } from "./types.js";

/**
 * 通过 npx 启动 codebase-memory-mcp managed runtime
 * 使用 stdio JSON-RPC 通信，完成 MCP initialize handshake
 */
export async function startManagedMcp(
  root: string,
  version: string,
  timeoutMs: number,
): Promise<McpLifecycleResult> {
  return new Promise((resolve) => {
    const child: ChildProcess = spawn(
      "npx",
      ["-y", `codebase-memory-mcp@${version}`],
      {
        cwd: root,
        stdio: ["pipe", "pipe", "pipe"],
        timeout: timeoutMs,
      },
    );

    let handshakeDone = false;

    const timeout = setTimeout(() => {
      if (!handshakeDone) {
        cleanup("UNAVAILABLE", "MCP 启动超时");
      }
    }, timeoutMs);

    const cleanup = (
      status: McpLifecycleResult["status"],
      message?: string,
    ) => {
      handshakeDone = true;
      clearTimeout(timeout);
      child.kill();
      resolve({
        provider: "codebase-memory-mcp",
        mode: "managed",
        status,
        pid: child.pid,
        message,
      });
    };

    child.on("error", (err) => {
      cleanup("FAILED", err.message);
    });

    child.on("exit", (code) => {
      if (!handshakeDone) {
        cleanup("FAILED", `MCP 进程退出，退出码: ${code}`);
      }
    });

    // MCP 协议：发送 initialize 请求
    const initRequest = JSON.stringify({
      jsonrpc: "2.0",
      method: "initialize",
      params: {
        protocolVersion: "2024-11-05",
        capabilities: {},
        clientInfo: { name: "sdd-harness", version: "0.1.0" },
      },
      id: 1,
    });

    child.stdin?.write(initRequest + "\n");

    // 简单握手检测：进程启动且无立即崩溃即视为 STARTED
    // 完整 MCP 初始化见后续 improvements
    setTimeout(() => {
      if (!handshakeDone) {
        cleanup("STARTED");
      }
    }, 1000);
  });
}

/** 停止 managed MCP 进程 */
export async function stopManagedMcp(
  result: McpLifecycleResult,
): Promise<void> {
  if (result.pid) {
    try {
      process.kill(result.pid, "SIGTERM");
    } catch {
      // 进程可能已退出
    }
  }
}
```

```typescript
import type { AgentTaskResult } from "./types/task-result.js";

/** 校验 AgentTaskResult 结构 */
export function validateTaskResult(data: unknown): AgentTaskResult {
  // 使用 zod 或手动校验
  if (!data || typeof data !== "object") {
    throw new Error("E_SCHEMA_VALIDATION_FAILED: AgentTaskResult 必须是对象");
  }
  const obj = data as Record<string, unknown>;
  if (obj.schemaVersion !== "1.0.0") {
    throw new Error("E_SCHEMA_VALIDATION_FAILED: schemaVersion 必须为 1.0.0");
  }
  if (!obj.taskId || typeof obj.taskId !== "string") {
    throw new Error("E_SCHEMA_VALIDATION_FAILED: taskId 必须是非空字符串");
  }
  if (!["DONE", "FAILED", "SKIPPED"].includes(obj.status as string)) {
    throw new Error("E_SCHEMA_VALIDATION_FAILED: status 必须为 DONE/FAILED/SKIPPED");
  }
  return obj as unknown as AgentTaskResult;
}
```
